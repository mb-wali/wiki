If / Else statements

#!/bin/python

n = int(raw_input())
if n % 2 == 1:
    print "Weird"
elif n % 2 == 0 and 2 <= n <= 5:
    print "Not Weird"
elif n % 2 == 0 and 6 <= n <= 20:
    print "Weird"
else:
    print "Not Weird"

    return False

def is_leap(year):
    return (year % 400 == 0) or (( year % 100 != 0) and (year % 4 == 0))

print is_leap(input())
